CREATE TABLE AVUKATLAR (
	ID INTEGER NOT NULL,
	AD VARCHAR(30),
	SIFRE VARCHAR(30),
	EMAIL VARCHAR(30),
	PRIMARY KEY (ID)
);



CREATE TABLE MUVEKKILLER (
	ID INTEGER NOT NULL, 
	AD VARCHAR(30), 
	EMAIL VARCHAR(30), 
	AVUKAT_ID INTEGER, 
	SOYAD VARCHAR(30), 
	TC BIGINT, 
	PRIMARY KEY (ID)
);



CREATE TABLE DURUSMALAR (
	DOSYANO INTEGER NOT NULL, 
	YER VARCHAR(30), 
	TARIH DATE, 
	TUR VARCHAR(30), 
	AVUKAT_ID INTEGER, 
	MUVEKKIL_ID INTEGER, 
	SAAT TIME, 
	PRIMARY KEY (DOSYANO)
);
